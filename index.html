const express = require('express'); const mongoose = require('mongoose'); const multer = require('multer'); const path = require('path'); require('dotenv').config(); const Multimedia = require('./models/Multimedia'); const app = express(); // Configuración de Multer para guardar archivos en una carpeta 'uploads' const storage = multer.diskStorage({ destination: './uploads/', filename: (req, file, cb) => { cb(null, Date.now() + path.extname(file.originalname)); } }); const upload = multer({ storage: storage }); app.use(express.json()); app.use(express.urlencoded({ extended: true })); app.use('/uploads', express.static('uploads')); // Para poder ver los archivos // Conexión DB mongoose.connect(process.env.MONGODB_URI) .then(() => console.log("Conectado a MongoDB")); // Ruta para recibir el formulario app.post('/api/multimedia', upload.fields([ { name: 'imagen', maxCount: 1 }, { name: 'audio', maxCount: 1 }, { name: 'video', maxCount: 1 } ]), async (req, res) => { try { const nuevoElemento = new Multimedia({ titulo: req.body.titulo, descripcion: req.body.descripcion, imagenUrl: req.files['imagen'] ? req.files['imagen'][0].path : '', audioUrl: req.files['audio'] ? req.files['audio'][0].path : '', videoUrl: req.files['video'] ? req.files['video'][0].path : '' }); await nuevoElemento.save(); res.send('Elemento guardado con éxito'); } catch (error) { res.status(500).send('Error al guardar: ' + error.message); } }); app.listen(3000, () => console.log('Servidor en puerto 3000'));
